// ── CORS headers for all responses ─────────────────────────────────────── function setCORSHeaders(output) { return output .setHeader("Access-Control-Allow-Origin", "*") .setHeader("Access-Control-Allow-Methods", "POST, GET, OPTIONS") .setHeader("Access-Control-Allow-Headers", "Content-Type"); } function doPost(e) { try { var raw = e.postData ? e.postData.contents : "{}"; var data = JSON.parse(raw); var ss = SpreadsheetApp.getActiveSpreadsheet(); var result = processData(ss, data); return setCORSHeaders( ContentService.createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON) ); } catch(err) { return setCORSHeaders( ContentService.createTextOutput(JSON.stringify({success:false,error:err.message})) .setMimeType(ContentService.MimeType.JSON) ); } } function doGet(e) { var p = e.parameter || {}; if(p.data) { // JSONP-style GET upload try { var data = JSON.parse(decodeURIComponent(p.data)); var ss = SpreadsheetApp.getActiveSpreadsheet(); var result = processData(ss, data); var json = JSON.stringify(result); if(p.callback) json = p.callback + "(" + json + ")"; return setCORSHeaders( ContentService.createTextOutput(json) .setMimeType(ContentService.MimeType.JAVASCRIPT) ); } catch(err) { return setCORSHeaders( ContentService.createTextOutput(JSON.stringify({success:false,error:err.message})) .setMimeType(ContentService.MimeType.JSON) ); } } return setCORSHeaders( ContentService.createTextOutput(JSON.stringify({status:"CG Inventory API running ✅"})) .setMimeType(ContentService.MimeType.JSON) ); } function processData(ss, data) { var type = data.type; var sheet = type === "inward" ? ss.getSheetByName("Inward") : type === "outward" ? ss.getSheetByName("Outward") : type === "hardware" ? ss.getSheetByName("Hardware_Register") : null; if(!sheet) return {success:false, error:"Sheet not found: "+type}; var rows = data.rows; if(!rows || rows.length===0) return {success:false, error:"No rows provided"}; // ── Build existing ID + Serial maps for duplicate checking ────────────── var lastRow = sheet.getLastRow(); var existingIds = {}; var existingSerials = {}; if(lastRow >= 4) { var existData = sheet.getRange(4, 1, lastRow-3, 13).getValues(); existData.forEach(function(r) { if(r[0]) existingIds[String(r[0]).trim()] = true; // Serial/IMEI is column H (index 7) if(r[7]) { String(r[7]).split(",").forEach(function(s){ s = s.trim(); if(s) existingSerials[s] = true; }); } }); } var added=0, skipped=0, duplicates=0, errors=[]; rows.forEach(function(row, i) { var rowId = row[0] ? String(row[0]).trim() : ""; var serial = row[7] ? String(row[7]).trim() : ""; // ── Duplicate Row ID check ─────────────────────────────────────────── if(rowId && existingIds[rowId]) { errors.push("Row "+(i+1)+": ⚠ Duplicate Inward ID \""+rowId+"\" — already exists in sheet. Skipped."); duplicates++; skipped++; return; } // ── Duplicate Serial check (outward: single; inward: comma-separated) ─ var serialList = serial ? serial.split(",").map(function(s){return s.trim();}) : []; var dupSerial = null; for(var j=0;j3){ hwSheet.getRange(4,2,hwSheet.getLastRow()-3,1).getValues() .forEach(function(r){if(r[0])existing.push(r[0].toString().trim());}); } var inData=inSheet.getRange(4,1,Math.max(1,inSheet.getLastRow()-3),13).getValues(); var newRows=[]; var ctr=hwSheet.getLastRow()-2; inData.forEach(function(row){ if(!row[0]) return; var serials=(row[7]||"").toString().split(","); serials.forEach(function(s){ s=s.trim(); if(!s||existing.indexOf(s)!==-1) return; ctr++; var id="HW-"+String(ctr).padStart(3,"0"); newRows.push([id,s,row[5],row[6],row[1],row[2],"Available","","","","",""]); existing.push(s); }); }); if(newRows.length>0) hwSheet.getRange(hwSheet.getLastRow()+1,1,newRows.length,12).setValues(newRows); } function autoProcessOutward(ss) { var outSheet=ss.getSheetByName("Outward"); var hwSheet=ss.getSheetByName("Hardware_Register"); if(!hwSheet||outSheet.getLastRow()<4) return; var outData=outSheet.getRange(4,1,outSheet.getLastRow()-3,13).getValues(); var hwSerials=hwSheet.getRange(4,2,Math.max(1,hwSheet.getLastRow()-3),7).getValues(); outData.forEach(function(oRow){ if(!oRow[7]) return; var serial=oRow[7].toString().trim(); hwSerials.forEach(function(hw,idx){ if(hw[0]===serial&&hw[5]!=="Issued"){ var r=idx+4; hwSheet.getRange(r,7).setValue("Issued"); hwSheet.getRange(r,8).setValue(oRow[2]); hwSheet.getRange(r,9).setValue(oRow[1]); hwSheet.getRange(r,10).setValue(oRow[3]); hwSheet.getRange(r,11).setValue(oRow[11]); } }); }); }
